For agent issuers

    

      npm install @ava-pay/agent and you can sign requests for either protocol. Three lines of code, plus one curl to register your key.
    

    
import { generateAgentKeyPair, signWithVisa } from '@ava-pay/agent';

const { privateKey } = generateAgentKeyPair();
const signed = signWithVisa({
  method: 'POST',
  url: 'https://shop.example.com/cart',
  body: JSON.stringify({ items: [...] }),
  agentId: 'agent_acme',
  privateKey,
  mandate: { /* user authorization */ },
});

await fetch(signed.url, { method: signed.method, headers: signed.headers, body: signed.body });
